// ====================================================== // Rentastic Stock Manager — Apps Script (Code.gs) // IMPORTANT: Deploy as Web App // Execute as: Me // Who has access: Anyone // ====================================================== const SHEET_ID = "1CHJsEvD-zNbzE89hgy0hI1URebNTMC5mQideIZUsy1Y"; const IN_OUT_SHEET = "In/Out"; const CLOSING_SHEET = "Daily Closing"; // ALL actions go through doGet — this avoids CORS preflight issues function doGet(e) { const p = e.parameter; const action = p.action || ""; const sheet = SpreadsheetApp.openById(SHEET_ID); try { // ---- ADD ENTRY (Stock In / Out) ---- if (action === "addEntry") { const ws = sheet.getSheetByName(IN_OUT_SHEET); const now = new Date(); const entryDate = p.date ? new Date(p.date) : now; const qty = parseInt(p.qty) || 0; // Append row: Timestamp | SKU | In/Out | Date | Quantity | Remarks ws.appendRow([now, p.sku, p.type, entryDate, qty, p.remarks || ""]); // Update Daily Closing tab updateClosingStock(sheet, p.sku, p.type, qty, entryDate); return json({ ok: true, message: "Entry saved." }); } // ---- GET CLOSING STOCK ---- if (action === "getClosing") { const ws = sheet.getSheetByName(CLOSING_SHEET); const data = ws.getDataRange().getValues(); return json({ ok: true, data: data }); } // ---- GET TODAY'S ENTRIES ---- if (action === "getToday") { const ws = sheet.getSheetByName(IN_OUT_SHEET); const data = ws.getDataRange().getValues(); const tz = Session.getScriptTimeZone(); const today = Utilities.formatDate(new Date(), tz, "yyyy-MM-dd"); const filtered = data.filter((row, i) => { if (i === 0) return true; if (!row[3]) return false; const d = Utilities.formatDate(new Date(row[3]), tz, "yyyy-MM-dd"); return d === today; }); return json({ ok: true, data: filtered }); } return json({ ok: false, error: "Unknown action: " + action }); } catch (err) { return json({ ok: false, error: err.message }); } } // Helper: return JSON with CORS headers function json(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } // Update or insert row in Daily Closing tab function updateClosingStock(sheet, sku, type, qty, date) { const ws = sheet.getSheetByName(CLOSING_SHEET); const tz = Session.getScriptTimeZone(); const dateStr = Utilities.formatDate(date, tz, "yyyy-MM-dd"); const data = ws.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (!data[i][0]) continue; const rowDate = Utilities.formatDate(new Date(data[i][0]), tz, "yyyy-MM-dd"); if (rowDate === dateStr && String(data[i][1]).trim() === String(sku).trim()) { const current = parseInt(data[i][2]) || 0; const newQty = type === "In" ? current + qty : current - qty; ws.getRange(i + 1, 3).setValue(newQty); return; } } // Not found — insert new row ws.appendRow([new Date(dateStr + "T00:00:00"), sku, type === "In" ? qty : -qty]); }
